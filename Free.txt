void Free(Wezel** korzen)
{
    if(korzen == NULL)
        return;
    if(*korzen == NULL)
        return;
    if((*korzen)->lewy != NULL)
        Free(&((*korzen)->lewy));
    if((*korzen)->prawy != NULL)
        Free(&((*korzen)->prawy));
    free((*korzen)->wycieczka->typ);
    free((*korzen)->wycieczka);
    free(*korzen);
    *korzen = NULL;
}
